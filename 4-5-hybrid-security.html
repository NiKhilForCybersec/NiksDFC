Onboard On-Premises Servers

Single Server Installation

PowerShell 
# Download and run Arc agent installer
$env:SUBSCRIPTION_ID = "your-subscription-id"
$env:RESOURCE_GROUP = "arc-servers-rg"
$env:TENANT_ID = "your-tenant-id"
$env:LOCATION = "eastus"

# Download the agent
Invoke-WebRequest -Uri "https://aka.ms/AzureConnectedMachineAgent" `
    -OutFile "$env:TEMP\AzureConnectedMachineAgent.msi"

# Install the agent
msiexec /i "$env:TEMP\AzureConnectedMachineAgent.msi" /quiet

# Connect to Azure
& "$env:ProgramFiles\AzureConnectedMachineAgent\azcmagent.exe" connect `
    --resource-group $env:RESOURCE_GROUP `
    --tenant-id $env:TENANT_ID `
    --location $env:LOCATION `
    --subscription-id $env:SUBSCRIPTION_ID
Bash 
# Download and install Arc agent
export SUBSCRIPTION_ID="your-subscription-id"
export RESOURCE_GROUP="arc-servers-rg"
export TENANT_ID="your-tenant-id"
export LOCATION="eastus"

# Download installation script
wget https://aka.ms/azcmagent -O ~/install_linux_azcmagent.sh

# Install agent
bash ~/install_linux_azcmagent.sh

# Connect to Azure
azcmagent connect \
    --resource-group $RESOURCE_GROUP \
    --tenant-id $TENANT_ID \
    --location $LOCATION \
    --subscription-id $SUBSCRIPTION_ID

At-Scale Onboarding

PowerShell 
# Generate service principal for at-scale deployment
$sp = New-AzADServicePrincipal -DisplayName "Arc-Onboarding-SP"
$credential = New-Object System.Management.Automation.PSCredential(
    $sp.AppId, 
    (ConvertTo-SecureString $sp.PasswordCredentials.SecretText -AsPlainText -Force)
)

# Assign Azure Connected Machine Onboarding role
New-AzRoleAssignment -ServicePrincipalName $sp.AppId `
    -RoleDefinitionName "Azure Connected Machine Onboarding" `
    -Scope "/subscriptions/$subscriptionId/resourceGroups/$resourceGroup"

# Export credentials for use in deployment scripts
@{
    ServicePrincipalId = $sp.AppId
    ServicePrincipalSecret = $sp.PasswordCredentials.SecretText
    TenantId = (Get-AzContext).Tenant.Id
} | ConvertTo-Json | Out-File "arc-credentials.json"

Enable Defender for Arc Servers

Deploy MDE Extension

Azure CLI 
# Enable Defender for Servers on subscription
az security pricing create \
    --name "VirtualMachines" \
    --tier "Standard" \
    --subplan "P2"

# Deploy MDE extension to Arc server
az connectedmachine extension create \
    --machine-name "arc-server-01" \
    --resource-group "arc-servers-rg" \
    --name "MDE.Windows" \
    --publisher "Microsoft.Azure.AzureDefenderForServers" \
    --type "MDE.Windows" \
    --settings '{}'

# For Linux
az connectedmachine extension create \
    --machine-name "arc-server-linux-01" \
    --resource-group "arc-servers-rg" \
    --name "MDE.Linux" \
    --publisher "Microsoft.Azure.AzureDefenderForServers" \
    --type "MDE.Linux" \
    --settings '{}'

Enable Auto-Provisioning

Azure CLI 
# Enable auto-provisioning for MDE
az security auto-provisioning-setting update \
    --name "default" \
    --auto-provision "On"

# Verify extension deployment
az connectedmachine extension list \
    --machine-name "arc-server-01" \
    --resource-group "arc-servers-rg" \
    -o table

Unified Monitoring

Query Hybrid Inventory

KQL 
// Get all Arc-enabled servers with health status
Resources
| where type == "microsoft.hybridcompute/machines"
| extend 
    Status = properties.status,
    OSType = properties.osType,
    AgentVersion = properties.agentVersion,
    LastStatusChange = properties.lastStatusChange
| project name, resourceGroup, location, Status, OSType, AgentVersion

// Check MDE extension status
Resources
| where type == "microsoft.hybridcompute/machines/extensions"
| where name contains "MDE"
| extend 
    MachineName = split(id, '/')[8],
    ProvisioningState = properties.provisioningState,
    Status = properties.instanceView.status.message
| project MachineName, name, ProvisioningState, Status

Security Posture Query

KQL 
// Compare Azure vs Arc server recommendations
SecurityResources
| where type == "microsoft.security/assessments"
| extend 
    ResourceType = tostring(properties.resourceDetails.ResourceType),
    IsArc = properties.resourceDetails.Id contains "hybridCompute",
    Status = tostring(properties.status.code),
    Severity = tostring(properties.metadata.severity)
| where Status == "Unhealthy"
| summarize Count = count() by IsArc, Severity
| order by IsArc, Severity
