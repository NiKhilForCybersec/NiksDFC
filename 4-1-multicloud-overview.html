Unified Security View

Query All Cloud Resources

KQL 
// Inventory across all clouds
SecurityResources
| where type == "microsoft.security/assessments"
| extend 
    CloudProvider = case(
        properties.resourceDetails.source == "Azure", "Azure",
        properties.resourceDetails.source == "Aws", "AWS",
        properties.resourceDetails.source == "Gcp", "GCP",
        "Unknown"
    ),
    Status = properties.status.code
| summarize 
    Healthy = countif(Status == "Healthy"),
    Unhealthy = countif(Status == "Unhealthy")
    by CloudProvider
KQL 
// Multi-cloud secure score
SecurityResources
| where type == "microsoft.security/securescores"
| extend 
    Score = properties.score.current,
    Max = properties.score.max,
    Percentage = properties.score.percentage
| project subscriptionId, Score, Max, Percentage

Cross-Cloud Attack Paths

KQL 
// Attack paths involving AWS or GCP resources
SecurityResources
| where type == "microsoft.security/attackpaths"
| extend 
    PathName = properties.attackPathDisplayName,
    RiskLevel = properties.riskLevel,
    EntryPoint = properties.entryPointEntityType,
    Target = properties.targetEntityType
| where properties.attackPathId contains "aws" or properties.attackPathId contains "gcp"
| project PathName, RiskLevel, EntryPoint, Target
