Hunting Queries

L3 Advanced 14 min read

Proactively hunt for threats using MDC data in Microsoft Sentinel. These hunting queries help identify suspicious patterns, misconfigurations, and potential attack chains before they trigger alerts.

Credential Access Hunting

Unusual Key Vault Access

KQL 
// Hunt for unusual Key Vault secret access patterns
AzureDiagnostics
| where ResourceType == "VAULTS"
| where OperationName == "SecretGet"
| summarize 
    AccessCount = count(),
    UniqueSecrets = dcount(id_s),
    Secrets = make_set(id_s, 10)
    by CallerIPAddress, identity_claim_upn_s, bin(TimeGenerated, 1h)
| where AccessCount > 20 or UniqueSecrets > 5
| project TimeGenerated, User = identity_claim_upn_s, CallerIPAddress, 
          AccessCount, UniqueSecrets, Secrets

Service Principal Secret Enumeration

KQL 
// Detect enumeration of service principal credentials
AuditLogs
| where OperationName has_any ("Add service principal credentials", 
                                "Update application ‚Äì Certificates and secrets management")
| where TimeGenerated > ago(7d)
| summarize 
    Operations = count(),
    AppIds = make_set(TargetResources[0].id)
    by InitiatedBy.user.userPrincipalName, bin(TimeGenerated, 1h)
| where Operations > 5
| project TimeGenerated, User = InitiatedBy_user_userPrincipalName, Operations, AppIds

Lateral Movement Hunting

Cross-Subscription Activity

KQL 
// Hunt for users accessing many subscriptions rapidly
AzureActivity
| where TimeGenerated > ago(24h)
| where ActivityStatusValue == "Success"
| summarize 
    SubscriptionCount = dcount(SubscriptionId),
    Subscriptions = make_set(SubscriptionId, 20),
    Operations = count()
    by Caller, bin(TimeGenerated, 1h)
| where SubscriptionCount > 5
| project TimeGenerated, Caller, SubscriptionCount, Operations, Subscriptions

VM to VM Network Connections

KQL 
// Detect unusual VM-to-VM connections (requires NSG flow logs)
AzureNetworkAnalytics_CL
| where TimeGenerated > ago(24h)
| where FlowType_s == "IntraVNet" or FlowType_s == "InterVNet"
| where DestPort_d in (22, 3389, 5985, 5986, 445)  // SSH, RDP, WinRM, SMB
| summarize 
    ConnectionCount = count(),
    DestPorts = make_set(DestPort_d),
    DestIPs = make_set(DestIP_s)
    by SrcIP_s, bin(TimeGenerated, 1h)
| where ConnectionCount > 10 or array_length(DestIPs) > 3
| project TimeGenerated, SourceVM = SrcIP_s, ConnectionCount, DestPorts, DestIPs

Persistence Hunting

New Role Assignments

KQL 
// Hunt for suspicious role assignments
AzureActivity
| where OperationNameValue == "Microsoft.Authorization/roleAssignments/write"
| where ActivityStatusValue == "Success"
| extend RoleDefinitionId = tostring(parse_json(Properties).requestbody)
| where RoleDefinitionId has_any ("Owner", "Contributor", "User Access Administrator")
| project TimeGenerated, Caller, SubscriptionId, 
          ResourceGroup, Resource, RoleDefinitionId

Automation Account Modifications

KQL 
// Detect changes to Automation runbooks (potential persistence)
AzureActivity
| where TimeGenerated > ago(7d)
| where ResourceProviderValue == "MICROSOFT.AUTOMATION"
| where OperationNameValue has_any ("runbooks/write", "webhooks/write", 
                                     "schedules/write", "credentials/write")
| project TimeGenerated, Caller, OperationNameValue, 
          Resource, ActivityStatusValue

Defense Evasion Hunting

Security Control Disabled

KQL 
// Hunt for Defender plans being disabled
AzureActivity
| where TimeGenerated > ago(7d)
| where ResourceProviderValue == "MICROSOFT.SECURITY"
| where OperationNameValue has "pricings/write"
| extend RequestBody = parse_json(Properties).requestbody
| where RequestBody has "Free"  // Plan being disabled
| project TimeGenerated, Caller, SubscriptionId, 
          Resource, RequestBody

Diagnostic Settings Removed

KQL 
// Detect removal of diagnostic settings (log tampering)
AzureActivity
| where TimeGenerated > ago(7d)
| where OperationNameValue has "diagnosticSettings/delete"
| project TimeGenerated, Caller, Resource, SubscriptionId, 
          ActivityStatusValue

NSG Rules Modified

KQL 
// Hunt for NSG rule changes that increase exposure
AzureActivity
| where TimeGenerated > ago(7d)
| where ResourceProviderValue == "MICROSOFT.NETWORK"
| where OperationNameValue has "networkSecurityGroups/securityRules/write"
| extend RequestBody = parse_json(Properties).requestbody
| where RequestBody has "0.0.0.0/0" or RequestBody has "*"
| project TimeGenerated, Caller, Resource, RequestBody

Exfiltration Hunting

Unusual Storage Access

KQL 
// Hunt for large data downloads from storage
StorageBlobLogs
| where TimeGenerated > ago(24h)
| where OperationName == "GetBlob"
| where ResponseBodySize > 100000000  // > 100MB
| summarize 
    TotalBytes = sum(ResponseBodySize),
    BlobCount = count(),
    Blobs = make_set(ObjectKey, 10)
    by CallerIpAddress, AccountName, bin(TimeGenerated, 1h)
| where TotalBytes > 1000000000  // > 1GB per hour
| extend TotalGB = round(TotalBytes / 1073741824.0, 2)
| project TimeGenerated, CallerIpAddress, AccountName, TotalGB, BlobCount

Anonymous Storage Access

KQL 
// Detect anonymous access to storage accounts
StorageBlobLogs
| where TimeGenerated > ago(7d)
| where AuthenticationType == "Anonymous"
| summarize 
    AccessCount = count(),
    TotalBytes = sum(ResponseBodySize),
    Containers = make_set(ObjectKey)
    by AccountName, CallerIpAddress
| where AccessCount > 10
| project AccountName, CallerIpAddress, AccessCount, TotalBytes, Containers

Posture Drift Hunting

New Unhealthy Recommendations

KQL 
// Hunt for newly unhealthy recommendations
SecurityRecommendation
| where TimeGenerated > ago(1d)
| where RecommendationState == "Unhealthy"
| where RecommendationSeverity in ("High", "Critical")
| summarize 
    NewFindings = count(),
    Resources = make_set(AssessedResourceId, 10)
    by RecommendationName, RecommendationSeverity
| order by NewFindings desc

Secure Score Drops

KQL 
// Hunt for significant secure score decreases
SecureScore_CL
| where TimeGenerated > ago(7d)
| summarize arg_max(TimeGenerated, *) by bin(TimeGenerated, 1d), SubscriptionId
| order by SubscriptionId, TimeGenerated asc
| serialize
| extend PreviousScore = prev(Score_d, 1)
| extend ScoreDrop = PreviousScore - Score_d
| where ScoreDrop > 3
| project TimeGenerated, SubscriptionId, CurrentScore = Score_d, 
          PreviousScore, ScoreDrop
