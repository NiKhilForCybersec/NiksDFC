Azure CLI - Defender for Cloud
$ # Enable Defender for Servers Plan 2 $ az security pricing create -n VirtualMachines --tier Standard $ # List all security alerts $ az security alert list --query "[?status=='Active']" $ # Get current Secure Score $ az security secure-score list --query "[].{Name:name, Score:currentScore}" $ # List security recommendations $ az security assessment list --query "[?status.code=='Unhealthy']"
KQL - Microsoft Sentinel
// High severity MDC alerts in last 24 hours SecurityAlert | where TimeGenerated > ago(24h) | where ProductName == "Azure Security Center" | where AlertSeverity == "High" | summarize count() by AlertName, bin(TimeGenerated, 1h)
